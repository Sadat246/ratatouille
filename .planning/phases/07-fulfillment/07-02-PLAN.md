  ```typescript
  import { randomInt } from "crypto";
  import { and, eq, inArray } from "drizzle-orm";

  import { fulfillments } from "@/db/schema";

  /**
   * 48 hours, in milliseconds. Pickup code expiry window per D-08.
   * Single source of truth — the choose-pickup service writes
   * `pickupCodeExpiresAt = new Date(Date.now() + PICKUP_CODE_TTL_MS)`,
   * and verify-pickup re-checks `pickupCodeExpiresAt > now()` against this
   * same window.
   */
  export const PICKUP_CODE_TTL_MS = 48 * 60 * 60 * 1000;

  /**
   * Status values that "lock" a code from re-use — only fulfillments in
   * one of these states can collide with a freshly generated code.
   * Picked-up / delivered / failed / cancelled rows release the code.
   */
  export const ACTIVE_PICKUP_STATUSES = [
    "awaiting_business",
    "ready_for_pickup",
  ] as const;

  /**
   * Returns a zero-padded 6-digit string drawn cryptographically from
   * [000000, 999999]. D-06 + D-09. Do NOT substitute Math.random.
   */
  export function generatePickupCode(): string {
    const n = randomInt(0, 1_000_000);
    return n.toString().padStart(6, "0");
  }

  /**
   * "482619" -> "482 619". D-06 display format. The single space separator
   * is intentional — easy to read aloud, fast for staff to type.
   */
  export function formatPickupCode(code: string): string {
    return `${code.slice(0, 3)} ${code.slice(3)}`;
  }

  /**
   * Minimal Drizzle-query surface this helper depends on. Declared
   * structurally so the unit test can pass a fake without importing
   * the real Neon-backed `db` instance.
   */
  export type PickupCodeQueryDb = {
    query: {
      fulfillments: {
        findFirst: (args: {
          columns: { id: true };
          where: ReturnType<typeof and>;
        }) => Promise<{ id: string } | undefined>;
      };
    };
  };

  /**
   * Issue a pickup code that does not collide with any currently-active
   * fulfillment. Retries up to 10 times — collision probability per attempt
   * is roughly (active_count / 1_000_000), so even at 10k active codes the
   * loop almost always succeeds on the first attempt. Throws (caller turns
   * into a 500) only in the genuinely-impossible exhaustion case so we
   * never silently issue a duplicate.
   */
  export async function generateUniquePickupCode(
    deps: { db: PickupCodeQueryDb },
  ): Promise<string> {
    for (let attempt = 0; attempt < 10; attempt++) {
      const code = generatePickupCode();
      const conflict = await deps.db.query.fulfillments.findFirst({
        columns: { id: true },
        where: and(
          eq(fulfillments.pickupCode, code),
          inArray(
            fulfillments.status,
            [...ACTIVE_PICKUP_STATUSES],
          ),
        ),
      });
      if (!conflict) {
        return code;
      }
    }
    throw new Error(
      "pickup code generation exhausted 10 attempts — active fulfillment count too high",
    );
  }
  ```

  Implementation notes:
  - `ACTIVE_PICKUP_STATUSES` is exported `as const` so callers (and tests) get a tuple-narrowed type and the test can iterate it.
  - `PickupCodeQueryDb` is a structural type, not `typeof db` — callers in Plan 07-04 will pass `{ db }` from `@/db/client`; the unit test passes a fake. This decouples the library from any concrete db client.
  - The error string starts with `pickup code generation exhausted` so Plan 07-04 / 07-05 can surface a clean 500 message without parsing internals.
  - No `import "server-only"` — `randomInt` works in Node test runners (Vitest); marking `server-only` would block the test from importing the module.

  import { describe, expect, it } from "vitest";

  import {
    ACTIVE_PICKUP_STATUSES,
    formatPickupCode,
    generatePickupCode,
    generateUniquePickupCode,
    PICKUP_CODE_TTL_MS,
    type PickupCodeQueryDb,
  } from "@/lib/fulfillment/pickup-code";

  function makeFakeDb(
    behaviour: (call: number) => { id: string } | undefined,
  ): { db: PickupCodeQueryDb; calls: number } {
    let calls = 0;
    const db: PickupCodeQueryDb = {
      query: {
        fulfillments: {
          findFirst: async () => {
            const result = behaviour(calls);
            calls += 1;
            return result;
          },
        },
      },
    };
    return {
      db,
      get calls() {
        return calls;
      },
    } as { db: PickupCodeQueryDb; calls: number };
  }

  describe("generatePickupCode", () => {
    it("returns exactly 6 characters", () => {
      for (let i = 0; i < 100; i++) {
        const code = generatePickupCode();
        expect(code).toHaveLength(6);
      }
    });

    it("uses only digits 0-9", () => {
      for (let i = 0; i < 100; i++) {
        expect(generatePickupCode()).toMatch(/^\d{6}$/);
      }
    });

    it("zero-pads small numbers (regression — random sample includes a low value)", () => {
      // Statistically over 100k samples we will hit something < 100000.
      let sawPadded = false;
      for (let i = 0; i < 100_000; i++) {
        const code = generatePickupCode();
        if (code.startsWith("0")) {
          sawPadded = true;
          expect(code).toHaveLength(6);
          break;
        }
      }
      expect(sawPadded).toBe(true);
    });
  });

  describe("formatPickupCode", () => {
    it("renders D-06 display format with a single space at index 3", () => {
      expect(formatPickupCode("482619")).toBe("482 619");
    });

    it("preserves leading zeros in the formatted output", () => {
      expect(formatPickupCode("004821")).toBe("004 821");
    });

    it("renders all-zero code", () => {
      expect(formatPickupCode("000000")).toBe("000 000");
    });
  });

  describe("PICKUP_CODE_TTL_MS", () => {
    it("equals 48 hours in milliseconds (D-08)", () => {
      expect(PICKUP_CODE_TTL_MS).toBe(48 * 60 * 60 * 1000);
      expect(PICKUP_CODE_TTL_MS).toBe(172_800_000);
    });
  });

  describe("ACTIVE_PICKUP_STATUSES", () => {
    it("locks pickup codes only against awaiting_business and ready_for_pickup", () => {
      expect([...ACTIVE_PICKUP_STATUSES]).toStrictEqual([
        "awaiting_business",
        "ready_for_pickup",
      ]);
    });

  describe("generateUniquePickupCode", () => {

      await expect(generateUniquePickupCode(fake)).rejects.toThrow(

      expect(fake.calls).toBe(10);
    });
  });
