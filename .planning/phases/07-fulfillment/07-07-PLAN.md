<objective>
Build the consumer-facing order detail page that owns the entire post-payment fulfillment surface for one settlement. The page is a server component that:

1. Authenticates as `consumer` (existing `requireCompletedRole`).
2. Fetches the fulfillment + settlement + consumer profile via the Plan 07-04 query helper (which enforces ownership).
3. Picks the right "state card" client component based on `fulfillment.status` — choice (pending_choice), pickup-code (awaiting_business / ready_for_pickup), delivery-tracking (delivery_requested / out_for_delivery), completed (picked_up / delivered), or failed (failed / cancelled).
4. Hands the consumer profile (for D-05 address pre-fill) and the formatted pickup code to the client cards.

The client orchestrator and the two interactive cards (Pickup, Delivery) drive the choice → POST → `router.refresh()` loop. All state-machine guarantees still live in the service + route layers; the UI only does presentation, fetch, and error display.

Purpose:
- D-04: post-payment confirmation screen presents two cards immediately. The buyer reaches this page from the My Bids "Won" CTA (Plan 07-09) or from a successful buyout flow (also Plan 07-09).
- D-05: delivery address pre-fills from `consumer_profiles` (already structured columns per OQ-2 resolution). The phone field is editable; the address is read-only in v1 (the consumer edits it in onboarding for v1; full inline edit is a v1.1 polish item — explicitly deferred in `<deferred>` block of CONTEXT).
- D-06 + D-08: pickup code displayed as `482 619` plus a 48-hour relative-countdown.
- D-12 + D-13: tracking link + status badge that auto-syncs as Uber webhooks land (the page is rendered on every navigation; no SSE in v1).

Output: One server-component page, three client components, one labels module. All type-check.
</objective>

  ```typescript
  "use client";

  import { useEffect, useState } from "react";

  import { SectionCard } from "@/components/auction/section-card";
  import { formatPickupCode } from "@/lib/fulfillment/pickup-code";

  type PickupCardProps = {
    pickupCode: string;
    pickupCodeExpiresAt: string; // ISO date from server
    business: {
      name: string;
      addressLine: string | null;
      pickupInstructions: string | null;
    };
  };

  export function PickupCard({
    pickupCode,
    pickupCodeExpiresAt,
    business,
  }: PickupCardProps) {
    const formatted = formatPickupCode(pickupCode);
    const expiresAtMs = new Date(pickupCodeExpiresAt).getTime();
    const [remainingLabel, setRemainingLabel] = useState(() =>
      formatRemaining(expiresAtMs - Date.now()),
    );

    useEffect(() => {
      const interval = setInterval(() => {
        setRemainingLabel(formatRemaining(expiresAtMs - Date.now()));
      }, 60_000);
      return () => clearInterval(interval);
    }, [expiresAtMs]);

    const expired = expiresAtMs <= Date.now();

    return (
      <SectionCard
        title="Show this code at the store"
        tone="border-[#d8e6de] bg-[rgba(241,248,244,0.92)] text-[#183227]"
      >
        <div className="flex flex-col items-start gap-4">
          <p className="text-[0.72rem] font-semibold uppercase tracking-[0.24em] text-[#476456]">
            Pickup code
          </p>
          <p
            className={`text-5xl font-semibold tracking-[0.18em] tabular-nums ${
              expired ? "text-[#9b6157] line-through" : "text-[#183227]"
            }`}
            aria-label={`Pickup code ${pickupCode.split("").join(" ")}`}
          >
            {formatted}
          </p>
          <p className="text-sm text-[#476456]">
            {expired
              ? "This code has expired. Contact the store to arrange pickup."
              : `Expires ${remainingLabel}.`}
          </p>
          <div className="mt-2 w-full rounded-[1.4rem] border border-[#cde1d7] bg-white/85 p-4">
            <p className="text-[0.68rem] font-semibold uppercase tracking-[0.24em] text-[#58806f]">
              {business.name}
            </p>
            {business.addressLine ? (
              <p className="mt-2 text-sm leading-6 text-[#173228]">
                {business.addressLine}
              </p>
            ) : null}
            {business.pickupInstructions ? (
              <p className="mt-2 text-sm leading-6 text-[#476456]">
                {business.pickupInstructions}
              </p>
            ) : null}
          </div>
        </div>
      </SectionCard>
    );
  }

  function formatRemaining(ms: number): string {
    if (ms <= 0) return "now";
    const totalMinutes = Math.floor(ms / 60_000);
    const hours = Math.floor(totalMinutes / 60);
    const minutes = totalMinutes % 60;
    if (hours <= 0) return `in ${minutes}m`;
    if (hours < 24) return `in ${hours}h ${minutes}m`;
    const days = Math.floor(hours / 24);
    const remainingHours = hours % 24;
    return `in ${days}d ${remainingHours}h`;
  }
  ```

  Notes:
  - Client component because of the `setInterval` countdown — server can't tick.
  - Recomputes every 60s, NOT every second; staff verification UX doesn't need sub-minute precision and the cheaper interval reduces battery / re-render churn.
  - The `aria-label` reads each digit individually so screen readers don't pronounce "482 619" as "four hundred eighty-two six hundred nineteen".
  - `tabular-nums` keeps the digit width fixed so the countdown doesn't reflow as it ticks down past 10h → 9h.
  - `formatRemaining` covers the 48-hour window cleanly: `48h 0m → 1m → now → expired branch`.
  - The business pickup-instructions render is null-safe (Phase 2 captured `pickupInstructions` as nullable).

  ```typescript
  "use client";

  import Link from "next/link";
  import { useRouter } from "next/navigation";
  import { useState } from "react";

  import { SectionCard } from "@/components/auction/section-card";
  import {
    consumerStatusLabel,
    type FulfillmentStatus,
  } from "@/lib/fulfillment/labels";

  type DeliveryAddress = {
    line1: string | null;
    line2: string | null;
    city: string | null;
    state: string | null;
    postalCode: string | null;
  };

  type DeliveryCardProps =
    | {
        mode: "request";
        fulfillmentId: string;
        address: DeliveryAddress;
        defaultPhone: string;
      }
    | {
        mode: "track";
        status: FulfillmentStatus;
        trackingUrl: string | null;
        recipientPhone: string | null;
        address: DeliveryAddress;
      };

  export function DeliveryCard(props: DeliveryCardProps) {
    if (props.mode === "request") {
      return <DeliveryRequestForm {...props} />;
    }
    return <DeliveryTrackingDisplay {...props} />;
  }

  function DeliveryRequestForm({
    fulfillmentId,
    address,
    defaultPhone,
  }: Extract<DeliveryCardProps, { mode: "request" }>) {
    const router = useRouter();
    const [phone, setPhone] = useState(defaultPhone);
    const [submitting, setSubmitting] = useState(false);
    const [error, setError] = useState<string | null>(null);

    async function handleSubmit(event: React.FormEvent<HTMLFormElement>) {
      event.preventDefault();
      setError(null);
      setSubmitting(true);
      try {
        const response = await fetch(
          `/api/fulfillment/${fulfillmentId}/choose-delivery`,
          {
            method: "POST",
            headers: { "content-type": "application/json" },
            body: JSON.stringify({ recipientPhone: phone.trim() }),
          },
        );
        const payload = (await response.json()) as
          | { ok: true }
          | { ok: false; error: { code: string; message: string } };

        if (!response.ok || !("ok" in payload) || !payload.ok) {
          const message =
            "error" in payload
              ? payload.error.message
              : "Could not request delivery. Try again.";
          setError(message);
          setSubmitting(false);
          return;
        }
        router.refresh();
      } catch {
        setError("Network error. Try again in a moment.");
        setSubmitting(false);
      }
    }

    return (
      <SectionCard
        title="Get it delivered"
        tone="border-[#d8e6de] bg-[rgba(241,248,244,0.92)] text-[#183227]"
      >
        <form onSubmit={handleSubmit} className="flex flex-col gap-4">
          <AddressBlock address={address} />
          <label className="flex flex-col gap-2">
            <span className="text-[0.72rem] font-semibold uppercase tracking-[0.24em] text-[#476456]">
              Recipient phone (driver contact)
            </span>
            <input
              type="tel"
              inputMode="tel"
              required
              minLength={10}
              value={phone}
              onChange={(event) => setPhone(event.target.value)}
              className="rounded-[1.2rem] border border-[#cde1d7] bg-white px-4 py-3 text-base text-[#173228]"
              placeholder="+1 555 555 0100"
            />
          </label>
          {error ? (
            <p className="rounded-[1rem] border border-[#ead1cb] bg-[rgba(255,241,237,0.88)] px-4 py-3 text-sm text-[#41231c]">
              {error}
            </p>
          ) : null}
          <button
            type="submit"
            disabled={submitting}
            className="rounded-full bg-[#1b3b33] px-6 py-3 text-sm font-semibold uppercase tracking-[0.18em] text-white disabled:opacity-60"
          >
            {submitting ? "Requesting driver…" : "Request delivery"}
          </button>
        </form>
      </SectionCard>
    );
  }

  function DeliveryTrackingDisplay({
    status,
    trackingUrl,
    recipientPhone,
    address,
  }: Extract<DeliveryCardProps, { mode: "track" }>) {
    const { label, tone } = consumerStatusLabel(status);
    return (
      <SectionCard
        title="Delivery in progress"
        tone="border-[#d8e6de] bg-[rgba(241,248,244,0.92)] text-[#183227]"
      >
        <div className="flex flex-col gap-4">
          <span

          <AddressBlock address={address} />

            </p>
          ) : null}
          {trackingUrl ? (
            <Link
              href={trackingUrl}
              target="_blank"
              rel="noreferrer noopener"
              className="rounded-full bg-[#1b3b33] px-6 py-3 text-center text-sm font-semibold uppercase tracking-[0.18em] text-white"

              Track your delivery
            </Link>
          ) : (
            <p className="text-sm text-[#476456]">

      </SectionCard>

  }

  function AddressBlock({ address }: { address: DeliveryAddress }) {
    const lines = [
      address.line1,
      address.line2,
      [address.city, address.state, address.postalCode]
        .filter(Boolean)
        .join(", "),
    ].filter((line): line is string => Boolean(line && line.length > 0));
    if (lines.length === 0) {
      return (
        <p className="rounded-[1rem] border border-[#ead1cb] bg-[rgba(255,241,237,0.88)] px-4 py-3 text-sm text-[#41231c]">

      );

    return (
      <div className="rounded-[1.2rem] border border-[#cde1d7] bg-white/85 p-4">
        <p className="text-[0.68rem] font-semibold uppercase tracking-[0.24em] text-[#58806f]">
          Delivering to
        </p>
        {lines.map((line) => (
          <p key={line} className="mt-1 text-sm text-[#173228]">
            {line}
          </p>
        ))}
      </div>
  ```
