  ```typescript
  import { NextResponse } from "next/server";

  import { authorizeApiRole } from "@/lib/auth/api";
  import {
    jsonFulfillmentError,
    toFulfillmentErrorResponse,
  } from "@/lib/fulfillment/http";
  import { verifyPickupCode } from "@/lib/fulfillment/service";

  export const runtime = "nodejs";

  type VerifyPickupBody = {
    submittedCode?: unknown;
  };

  export async function POST(
    request: Request,
    context: { params: Promise<{ id: string }> },
  ) {
    const authorization = await authorizeApiRole("business");
    if (!authorization.ok) {
      return NextResponse.json(authorization.body, {
        status: authorization.status,
      });
    }

    let body: VerifyPickupBody;
    try {
      body = (await request.json()) as VerifyPickupBody;
    } catch {
      return jsonFulfillmentError(
        "VALIDATION_FAILED",
        "Request body must be JSON.",
        400,
      );
    }

    const submittedCode =
      typeof body.submittedCode === "string" ? body.submittedCode.trim() : "";

    try {
      const { id: fulfillmentId } = await context.params;
      const result = await verifyPickupCode({
        fulfillmentId,
        businessUserId: authorization.session.user.id,
        submittedCode,
      });

      return NextResponse.json({
        ok: true,
        fulfillmentId: result.fulfillmentId,
        status: result.status,
        deliveredAt: result.deliveredAt.toISOString(),
      });
    } catch (error) {
      return toFulfillmentErrorResponse(error);
    }
  }
  ```

  ```typescript
  import { NextResponse } from "next/server";

  import { getOptionalEnv } from "@/lib/env";
  import {
    markUberEventProcessed,
    wasUberEventProcessed,
  } from "@/lib/fulfillment/idempotency";
  import { applyUberStatusUpdate } from "@/lib/fulfillment/service";
  import { UBER_DIRECT_ENABLED } from "@/lib/fulfillment/uber-direct";
  import { verifyUberSignature } from "@/lib/fulfillment/uber-signature";

  export const runtime = "nodejs";

  type UberWebhookPayload = {
    id?: unknown;
    kind?: unknown;
    status?: unknown;
    delivery_id?: unknown;
  };

  export async function POST(request: Request) {
    const rawBody = await request.text();
    const signature = request.headers.get("x-postmates-signature") ?? "";

    if (UBER_DIRECT_ENABLED) {
      const secret = getOptionalEnv("UBER_DIRECT_WEBHOOK_SECRET") ?? "";
      if (!secret || !verifyUberSignature(rawBody, signature, secret)) {
        return NextResponse.json(
          { ok: false, error: { code: "INVALID_SIGNATURE", message: "Signature mismatch." } },
          { status: 401 },
        );
      }
    } else {
      // D-02 stub mode: no secret to verify against; log + accept so the
      // demo can replay synthesized events from a fixture without 401s.
      console.warn(
        "[uber-direct/webhook] UBER_DIRECT_ENABLED=false; skipping signature verification (D-02 stub mode)",
      );
      return NextResponse.json({ ok: true, skipped: true });
    }

    let payload: UberWebhookPayload;
    try {
      payload = JSON.parse(rawBody) as UberWebhookPayload;
    } catch {
      // Malformed JSON after a valid signature is suspicious; refuse without
      // logging the body (could contain PII).
      return NextResponse.json(
        { ok: false, error: { code: "INVALID_PAYLOAD", message: "Body is not JSON." } },
        { status: 400 },
      );
    }

    const eventId = typeof payload.id === "string" ? payload.id : "";
    const eventType = typeof payload.kind === "string" ? payload.kind : "";
    const uberStatus = typeof payload.status === "string" ? payload.status : "";
    const deliveryReferenceId =
      typeof payload.delivery_id === "string" ? payload.delivery_id : "";

    // We only care about delivery_status events. Other event kinds get
    // a 200 ack so Uber stops retrying — but we don't dispatch.
    if (eventType !== "event.delivery_status") {
      return NextResponse.json({ ok: true, applied: false, reason: "ignored_event_kind" });
    }

    if (!eventId || !deliveryReferenceId || !uberStatus) {
      return NextResponse.json(
        { ok: false, error: { code: "MISSING_FIELDS", message: "Required fields absent." } },
        { status: 400 },
      );
    }

    // Idempotency check BEFORE any state mutation. Uber retries up to 4×.
    if (await wasUberEventProcessed(eventId)) {
      return NextResponse.json({ ok: true, applied: false, reason: "already_processed" });
    }

    try {
      const result = await applyUberStatusUpdate({
        deliveryReferenceId,
        uberStatus,
        eventId,
        eventType,
      });

      // Persist the dedup row AFTER the service applies the transition so
      // a failure during dispatch (e.g. transient DB error) leaves the
      // event un-marked and Uber will retry.
      await markUberEventProcessed(eventId, eventType);

      return NextResponse.json({
        ok: true,
        applied: result.applied,
        newStatus: result.newStatus,
      });
    } catch (error) {
      console.error("[uber-direct/webhook] dispatch failed", error);
      return NextResponse.json(
        { ok: false, error: { code: "DISPATCH_FAILED", message: "Could not apply status update." } },
        { status: 500 },
      );
    }
  }
  ```

  Notes:
  - Reads `request.text()` BEFORE any JSON parse — same pattern as `app/api/webhooks/stripe/route.ts`. The HMAC is computed over the raw bytes; round-tripping through `JSON.parse → JSON.stringify` would alter whitespace and break verification.
  - Header is `x-postmates-signature` (Uber kept the legacy Postmates header name; verified in RESEARCH).
  - Stub-mode short-circuits with a 200 + `skipped:true` response — the `console.warn` is a deliberate operator-visible breadcrumb, not an error.
  - Idempotency check happens BEFORE applying the status. If `applyUberStatusUpdate` throws, `markUberEventProcessed` is NOT called, so Uber's retry will re-attempt. (If the transition succeeded but `markUberEventProcessed` throws, Uber's retry will produce a no-op transition because the service layer also guards against backwards transitions — defence in depth.)
  - Unknown event kinds are 200-acked so Uber stops retrying; we never want to keep them in Uber's retry queue.
  - Required `runtime = "nodejs"` because `verifyUberSignature` uses Node `crypto.createHmac` and the service layer opens an interactive Postgres transaction.

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Browser (business staff) → verify-pickup route | Authenticated session required; ownership re-checked in service via business membership join |
| Internet → uber-direct webhook | Unauthenticated transport — gated by HMAC-SHA256 signature in real mode; permissive in D-02 stub mode |
| Webhook event payload → service layer | Untrusted JSON; explicit field narrowing + service-layer state-machine guard |

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-07-30 | ASVS V4 (Access Control) | verify-pickup route | mitigate | Role-gated by `authorizeApiRole("business")`. Service layer joins `business_memberships` to enforce that the calling user belongs to the row's `settlements.businessId`; mismatch → `FULFILLMENT_NOT_FOUND` (404 collapse) so a logged-in business cannot probe other shops' fulfillments. |

| T-07-32 | ASVS V13 (API & Web Service) | uber-direct webhook signature | mitigate | When `UBER_DIRECT_ENABLED === true`, missing/wrong `x-postmates-signature` returns 401 BEFORE any DB hit and BEFORE parsing the body for fields. `verifyUberSignature` (Plan 07-03) uses Node `crypto.createHmac("sha256", secret)` and a constant-time compare. |
| T-07-33 | ASVS V13 (Logging) | uber-direct webhook stub mode | accept | `UBER_DIRECT_ENABLED === false` skips signature verification by design (D-02 demo posture); a `console.warn` records every accepted event so the operator sees it. | T-07-35 | ASVS V7 (Error handling) | uber-direct dispatch failure | mitigate | Service-layer throw → 500 response with `DISPATCH_FAILED` (no payload echo); Uber's retry policy (4× exponential backoff per RESEARCH) covers transient failures.
