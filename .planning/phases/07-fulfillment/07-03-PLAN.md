<objective>
Build the four pure-function libraries that wrap every external Uber Direct concern: SDK client + stub fallback (`uber-direct.ts`), status projection (`status-map.ts`), address serialization (`address-format.ts`), and webhook signature verification (`uber-signature.ts`). All four are isolated from the database and from Next.js request internals so each ships with its own Vitest spec, and the service layer (Plan 07-04) and route handlers (Plans 07-05 / 07-06) consume them as named imports without further plumbing.

Purpose: D-01 (real Uber API) and D-02 (stub fallback) collapse into a single env-gated factory inside `uber-direct.ts`; both code paths return the same `{ deliveryReferenceId, trackingUrl }` shape so the calling service layer is mode-agnostic. The status map (RESEARCH Uber Status → Ratatouille Schema Status Map) is the single source of truth that both the webhook handler and any future polling job consume. The address formatter neutralises Pitfall §7 (Uber rejects plain-string addresses). The signature verifier is the only piece of crypto that runs on the webhook hot path; isolating it makes it trivially testable with a known-good HMAC.

Output: Four library modules under `lib/fulfillment/`, four Vitest specs under `tests/lib/fulfillment/`, all type-checking and passing tests; ready for Plan 07-04 to compose them into the fulfillment service layer.
</objective>

   * keys positionally. Stable order also makes test snapshots deterministic.
   */

    streetLine1: string;
    streetLine2?: string | null;
    city: string;
    state: string;
    postalCode: string;
    countryCode: string;
  };

  /**
   * Returns the JSON-encoded address string Uber Direct's `pickup_address`
   * and `dropoff_address` fields require. Combines streetLine1 + streetLine2
   * into a `street_address` array so multi-line addresses survive intact.

    const streetLines = [input.streetLine1.trim()];
    if (input.streetLine2 && input.streetLine2.trim() !== "") {
      streetLines.push(input.streetLine2.trim());

      street_address: streetLines,
      city: input.city.trim(),
      state: input.state.trim(),
      zip_code: input.postalCode.trim(),

  ```typescript
  import "server-only";
  import { createHmac, timingSafeEqual } from "crypto";

  /**
   * Verify an Uber Direct webhook payload. Algorithm and header are
   * fixed by Uber per developer.uber.com/docs/deliveries/guides/webhooks
   * (HMAC-SHA256 of the raw body, hex digest, header `x-uber-signature`).
   *
   * Uses `timingSafeEqual` to defeat trivial timing-based signature
   * extraction. Returns `false` (never throws) so the route handler can
   * convert the result into a 400 without a try/catch.
   */
  export function verifyUberSignature(
    rawBody: string,
    signature: string | null | undefined,
    secret: string,
  ): boolean {
    if (!signature || signature.length === 0) {
      return false;
    }
    const computed = createHmac("sha256", secret)
      .update(rawBody, "utf8")
      .digest("hex");

    // Bail out on length mismatch BEFORE timingSafeEqual — it throws on
    // unequal-length buffers.
    if (computed.length !== signature.length) {
      return false;
    }

    try {
      return timingSafeEqual(
        Buffer.from(computed, "utf8"),
        Buffer.from(signature, "utf8"),
      );
    } catch {
      return false;
    }
  }
  ```

  Implementation notes:
  - `import "server-only"` — webhook secrets must never reach a client bundle.
  - `timingSafeEqual` is the standard Node defense for HMAC comparison; Stripe's `constructEvent` does the same internally.
  - Returning `false` (not throwing) keeps the route-handler control flow flat (`if (!verifyUberSignature(...)) return 400`).

  import "server-only";

  import { hasEnv } from "@/lib/env";

  /**
   * D-02 gate. When ANY of the three credentials is missing, the entire
   * Uber Direct integration silently falls back to the stub layer — both
   * `createUberDelivery` and `cancelUberDelivery` synthesise sandbox-shaped
   * responses, and the webhook route (Plan 07-06) skips signature
   * verification because there is no secret to verify against.
   */
  export const UBER_DIRECT_ENABLED = hasEnv(
    "UBER_DIRECT_CLIENT_ID",
    "UBER_DIRECT_CLIENT_SECRET",
    "UBER_DIRECT_CUSTOMER_ID",
  );

  export type CreateUberDeliveryInput = {
    /** Display name shown to the courier at the storefront. */
    pickupName: string;
    /** JSON-stringified address from `formatBusinessPickupAddress`. */
    pickupAddress: string;
    pickupPhone: string;
    /** Optional staff-facing pickup instructions. */
    pickupInstructions?: string | null;
    dropoffName: string;
    /** JSON-stringified address from `formatConsumerDeliveryAddress`. */
    dropoffAddress: string;
    dropoffPhone: string;
    /** Listing title — Uber requires at least one manifest item. */
    itemName: string;
    /** Settlement gross amount in cents — for Uber's manifest, not for charging. */
    itemPriceCents: number;
    /**
     * `fulfillment.id` UUID — Uber treats this as the idempotency key on
     * the create-delivery side. Two calls with the same external_order_id
     * within the dedup window return the SAME delivery, not a duplicate.
     */
    externalOrderId: string;
  };

  export type CreateUberDeliveryResult = {
    deliveryReferenceId: string;
    trackingUrl: string;
    /** Initial Uber status; webhook handler (Plan 07-06) will overwrite. */
    initialStatus: string;
    /** Whether the result came from the stub layer (D-02). */
    stub: boolean;
  };

  /**
   * Create a delivery against Uber Direct, or return a sandbox-shaped stub
   * response when credentials are unavailable (D-02).
   *
   * Stub responses are demo-safe: tracking URL points to a placeholder Uber
   * Eats orders page, the deliveryReferenceId is uniquely synthetic, and
   * the initialStatus mirrors a real `pending` response. The Plan 07-06
   * webhook route is never invoked in stub mode (Uber never knows about
   * the order), so consumer-side status will remain at `delivery_requested`
   * — for a pitch demo we recommend visiting the consumer order page
   * after the choose-delivery POST and showing the "Track your delivery"
   * link as the demonstrable artefact.
   */
  export async function createUberDelivery(
    input: CreateUberDeliveryInput,
  ): Promise<CreateUberDeliveryResult> {
    if (!UBER_DIRECT_ENABLED) {
      const stamp = Date.now();
      return {
        deliveryReferenceId: `stub_del_${stamp}`,
        trackingUrl: `https://www.ubereats.com/orders/stub-${stamp}`,
        initialStatus: "pending",
        stub: true,
      };
    }

    // Real path — dynamic import keeps the SDK out of any cold-start
    // critical path until UBER_DIRECT_ENABLED has been observed true at
    // request time. Avoids loading SDK code (and its transitive deps) in
    // stub-mode deployments.
    const { getAccessToken } = await import("uber-direct/auth");
    const { createDeliveriesClient } = await import("uber-direct/deliveries");

    const token = await getAccessToken();
    const client = createDeliveriesClient(token);

    const result = await client.createDelivery({
      pickup_name: input.pickupName,
      pickup_address: input.pickupAddress,
      pickup_phone_number: input.pickupPhone,
      pickup_notes: input.pickupInstructions ?? undefined,
      dropoff_name: input.dropoffName,
      dropoff_address: input.dropoffAddress,
      dropoff_phone_number: input.dropoffPhone,
      manifest_items: [
        {
          name: input.itemName,
          quantity: 1,
          size: "small",
          price: input.itemPriceCents,
        },
      ],
      external_order_id: input.externalOrderId,
      // Pitfall §3: sandbox deliveries stall in `pending` forever
      // without the robo-courier flag; harmless in production because
      // production credentials reject `testSpecifications` cleanly.
      testSpecifications: {
        roboCourierSpecification: { mode: "auto" },
      },
    });

    return {
      deliveryReferenceId: result.id,
      trackingUrl: result.tracking_url,
      initialStatus: result.status ?? "pending",
      stub: false,
    };
  }

  /**
   * Cancel an in-flight Uber Direct delivery. Stub mode is a no-op
   * `{ ok: true, stub: true }` — useful only if a future plan exposes a
   * cancel button. Phase 7 itself does NOT call this in any normal flow
   * (D-14 says no auto-refund on failure), but exporting it keeps the
   * SDK contact surface in one file.
   */
  export async function cancelUberDelivery(
    deliveryReferenceId: string,
  ): Promise<{ ok: boolean; stub: boolean }> {

  import { createHmac } from "crypto";

  import { describe, expect, it } from "vitest";

  import { verifyUberSignature } from "@/lib/fulfillment/uber-signature";

  function sign(body: string, secret: string): string {
    return createHmac("sha256", secret).update(body, "utf8").digest("hex");
  }

  describe("verifyUberSignature", () => {
    const secret = "uber-test-signing-key";
    const body = JSON.stringify({
      kind: "event.delivery_status",
      delivery_id: "del_abc123",
      status: "delivered",
    });

    it("accepts a correctly-computed signature", () => {
      expect(verifyUberSignature(body, sign(body, secret), secret)).toBe(true);
    });

    it("rejects a tampered body", () => {
      const tampered = body.replace("delivered", "canceled");
      expect(verifyUberSignature(tampered, sign(body, secret), secret)).toBe(false);
    });

    it("rejects a wrong secret", () => {
      expect(verifyUberSignature(body, sign(body, "different"), secret)).toBe(false);
    });

    it("rejects an empty signature", () => {
      expect(verifyUberSignature(body, "", secret)).toBe(false);
      expect(verifyUberSignature(body, null, secret)).toBe(false);
      expect(verifyUberSignature(body, undefined, secret)).toBe(false);
    });

    it("rejects a signature of the wrong length without throwing", () => {
      expect(verifyUberSignature(body, "abcdef", secret)).toBe(false);
    });
  });
