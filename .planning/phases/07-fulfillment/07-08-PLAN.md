  ```typescript
  "use client";

  import Link from "next/link";
  import { useRouter } from "next/navigation";
  import { useState } from "react";

  import {
    businessStatusLabel,
    type FulfillmentStatus,
    type LabelTone,
  } from "@/lib/fulfillment/labels";

  type RowProps = {
    fulfillmentId: string;
    status: FulfillmentStatus;
    mode: "pickup" | "delivery";
    listingTitle: string;
    buyerDisplayName: string;
    pickupCodeExpiresAt: string | null; // ISO
    deliveredAt: string | null;          // ISO
    trackingUrl: string | null;
    recipientPhone: string | null;
    createdAt: string;                   // ISO
  };

  export function BusinessFulfillmentRow(props: RowProps) {
    const { label, tone } = businessStatusLabel(props.status);
    return (
      <li className="rounded-[1.6rem] border border-[#ecdac7] bg-white/90 p-4 shadow-[0_8px_30px_rgba(78,47,28,0.05)]">
        <div className="flex flex-wrap items-start justify-between gap-3">
          <div className="min-w-0 flex-1">
            <p className="truncate text-base font-semibold tracking-[-0.02em] text-[#2a2018]">
              {props.listingTitle}
            </p>
            <p className="mt-1 text-sm text-[#8b6a53]">
              For {props.buyerDisplayName} • {props.mode === "pickup" ? "Pickup" : "Delivery"}
            </p>
          </div>
          <StatusBadge label={label} tone={tone} />
        </div>

        {props.mode === "pickup" && isPickupActive(props.status) ? (
          <PickupVerifyForm
            fulfillmentId={props.fulfillmentId}
            pickupCodeExpiresAt={props.pickupCodeExpiresAt}
          />
        ) : null}

        {props.mode === "delivery" && (props.status === "delivery_requested" || props.status === "out_for_delivery") ? (
          <DeliveryActions trackingUrl={props.trackingUrl} recipientPhone={props.recipientPhone} />
        ) : null}

        {props.status === "failed" ? (
          <FailedDeliveryActions recipientPhone={props.recipientPhone} />
        ) : null}

        {props.status === "picked_up" || props.status === "delivered" ? (
          <p className="mt-3 text-xs text-[#8b6a53]">
            Completed {props.deliveredAt ? new Date(props.deliveredAt).toLocaleString() : "—"}
          </p>
        ) : null}
      </li>
    );
  }

  function isPickupActive(status: FulfillmentStatus): boolean {
    return status === "awaiting_business" || status === "ready_for_pickup";
  }

  function PickupVerifyForm({
    fulfillmentId,
    pickupCodeExpiresAt,
  }: {
    fulfillmentId: string;
    pickupCodeExpiresAt: string | null;
  }) {
    const router = useRouter();
    const [code, setCode] = useState("");
    const [submitting, setSubmitting] = useState(false);
    const [error, setError] = useState<string | null>(null);

    const expired = pickupCodeExpiresAt
      ? new Date(pickupCodeExpiresAt).getTime() <= Date.now()
      : false;

    async function handleSubmit(event: React.FormEvent<HTMLFormElement>) {
      event.preventDefault();
      setError(null);
      setSubmitting(true);
      try {
        const response = await fetch(
          `/api/fulfillment/${fulfillmentId}/verify-pickup`,
          {
            method: "POST",
            headers: { "content-type": "application/json" },
            body: JSON.stringify({ submittedCode: code.trim() }),
          },
        );
        const payload = (await response.json()) as
          | { ok: true; status: string }
          | { ok: false; error: { code: string; message: string } };

        if (!response.ok || !("ok" in payload) || !payload.ok) {
          const message =
            "error" in payload ? payload.error.message : "Could not verify code.";
          setError(message);
          setSubmitting(false);
          return;
        }
        setCode("");
        router.refresh();
      } catch {
        setError("Network error. Try again.");
        setSubmitting(false);
      }
    }

    if (expired) {
      return (
        <p className="mt-3 rounded-[1rem] border border-[#ead1cb] bg-[rgba(255,241,237,0.88)] px-3 py-2 text-xs text-[#41231c]">
          Code expired. Buyer must contact you to arrange pickup.
        </p>
      );
    }

    return (
      <form onSubmit={handleSubmit} className="mt-3 flex flex-wrap items-center gap-2">
        <label className="sr-only" htmlFor={`code-${fulfillmentId}`}>
          Pickup code
        </label>
        <input
          id={`code-${fulfillmentId}`}
          inputMode="numeric"
          pattern="[0-9]*"
          maxLength={6}
          minLength={6}
          required
          autoComplete="off"
          value={code}
          onChange={(event) => setCode(event.target.value.replace(/\D/g, ""))}
          placeholder="6-digit code"
          className="w-36 rounded-full border border-[#ecdac7] bg-white px-4 py-2 text-base tabular-nums tracking-[0.18em] text-[#2a2018]"
        />
        <button
          type="submit"
          disabled={submitting || code.length !== 6}
          className="rounded-full bg-[#3d2618] px-4 py-2 text-xs font-semibold uppercase tracking-[0.18em] text-white disabled:opacity-60"
        >
          {submitting ? "Verifying…" : "Verify pickup"}
        </button>
        {error ? (
          <span className="basis-full text-xs text-[#41231c]">{error}</span>
        ) : null}
      </form>
    );
  }

  function DeliveryActions({
    trackingUrl,
    recipientPhone,
  }: {
    trackingUrl: string | null;
    recipientPhone: string | null;
  }) {
    return (
      <div className="mt-3 flex flex-wrap gap-2 text-xs">
        {trackingUrl ? (
          <Link
            href={trackingUrl}
            target="_blank"
            rel="noreferrer noopener"
            className="rounded-full border border-[#ecdac7] px-3 py-1.5 font-semibold uppercase tracking-[0.18em] text-[#3d2618]"
          >
            Driver tracking
          </Link>
        ) : null}
        {recipientPhone ? (
          <span className="rounded-full bg-[rgba(247,238,225,0.9)] px-3 py-1.5 text-[#3d2618]">
            Buyer: {recipientPhone}
          </span>
        ) : null}
      </div>
    );
  }

  function FailedDeliveryActions({ recipientPhone }: { recipientPhone: string | null }) {
    return (
      <div className="mt-3 rounded-[1rem] border border-[#ead1cb] bg-[rgba(255,241,237,0.88)] px-3 py-2 text-xs text-[#41231c]">
        Delivery failed. {recipientPhone ? `Contact buyer at ${recipientPhone} to arrange manual pickup.` : "Contact the buyer to arrange manual pickup."}
      </div>
    );
  }

  function StatusBadge({ label, tone }: { label: string; tone: LabelTone }) {
    return (
      <span
        className={`inline-flex shrink-0 rounded-full border px-3 py-1 text-[0.66rem] font-semibold uppercase tracking-[0.22em] ${toneToClass(tone)}`}
      >
        {label}
      </span>
    );
  }

  function toneToClass(tone: LabelTone): string {
    switch (tone) {
      case "success":
        return "border-[#b9d8c7] bg-[rgba(214,236,224,0.92)] text-[#183227]";
      case "info":
        return "border-[#bcd4ea] bg-[rgba(225,237,250,0.92)] text-[#1b3a55]";
      case "warning":
        return "border-[#ead4b1] bg-[rgba(252,239,217,0.92)] text-[#4d3416]";
      case "danger":
        return "border-[#ead1cb] bg-[rgba(255,241,237,0.92)] text-[#41231c]";
      case "neutral":
      default:
        return "border-[#d4d8df] bg-[rgba(238,240,244,0.92)] text-[#2c2f36]";
    }
  }
  ```
