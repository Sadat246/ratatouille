    export function computeHaversine(lat1: number, lng1: number, lat2: number, lng2: number): number {
      const EARTH_RADIUS_MILES = 3959;
      const inner =
        Math.cos((lat1 * Math.PI) / 180) *
        Math.cos((lat2 * Math.PI) / 180) *
        Math.cos(((lng2 - lng1) * Math.PI) / 180) +
        Math.sin((lat1 * Math.PI) / 180) *

    }

    ```typescript
    import { NextResponse } from "next/server";
    import { db } from "@/db/client";
    import { authorizeApiRole } from "@/lib/auth/api";
    import { getAuctionFeed } from "@/lib/auctions/queries";
    import { listingCategoryValues } from "@/lib/listings/categories";
    import { sweepOverdueAuctions } from "@/lib/auctions/service";
    import { AUCTION_SWEEP_BATCH_SIZE } from "@/lib/auctions/pricing";

    export const runtime = "nodejs";

    const VALID_SORT_BY = ["ending_soon", "nearest", "lowest_price"] as const;
    type SortBy = (typeof VALID_SORT_BY)[number];

    export async function GET(request: Request) {
      const authorization = await authorizeApiRole("consumer");
      if (!authorization.ok) {
        return NextResponse.json(authorization.body, { status: authorization.status });
      }

      try {
        const { searchParams } = new URL(request.url);
        const offset = Math.max(0, Number(searchParams.get("offset") ?? "0"));
        const LIMIT = 12;
        const rawSortBy = searchParams.get("sortBy") ?? "ending_soon";
        const sortBy: SortBy = VALID_SORT_BY.includes(rawSortBy as SortBy)
          ? (rawSortBy as SortBy)
          : "ending_soon";
        const rawCategories = searchParams.getAll("category");
        const categories = rawCategories.filter((c) =>
          (listingCategoryValues as readonly string[]).includes(c)
        );

        // SECURITY: read consumer lat/lng from DB — NEVER from request params
        const profile = await db.query.consumerProfiles.findFirst({
          columns: { latitude: true, longitude: true },
          where: (table, operators) =>
            operators.eq(table.userId, authorization.session.user.id),
        });

        if (!profile) {
          return NextResponse.json({ ok: false, error: "No consumer profile found" }, { status: 400 });
        }

        await sweepOverdueAuctions(AUCTION_SWEEP_BATCH_SIZE);

        const items = await getAuctionFeed({
          lat: profile.latitude,
          lng: profile.longitude,
          sortBy,
          categories,
          limit: LIMIT + 1, // fetch one extra to detect hasMore
          offset,
          viewerUserId: authorization.session.user.id,
        });

        const hasMore = items.length > LIMIT;
        return NextResponse.json({
          ok: true,
          items: items.slice(0, LIMIT),
          nextOffset: hasMore ? offset + LIMIT : null,
          hasMore,
        });
      } catch (error) {
        console.error("[feed] GET error", error);
        return NextResponse.json({ ok: false, error: "Internal error" }, { status: 500 });
      }
    }
    ```

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-05-01 | Tampering | `GET /api/auctions/feed` | mitigate | Consumer lat/lng is read exclusively from `db.query.consumerProfiles` keyed on `authorization.session.user.id` — never from request query params. Prevents location spoofing to see listings outside consumer's area. |
| T-05-02 | Tampering | `sortBy` query param | mitigate | Validate against `VALID_SORT_BY = ["ending_soon","nearest","lowest_price"]` whitelist; invalid values fall back to `"ending_soon"`. |
| T-05-03 | Tampering | `category` query params | mitigate | Filter raw categories through `listingCategoryValues` whitelist before passing to `inArray()`. Prevents arbitrary strings reaching the SQL query. |
| T-05-04 | DoS | `offset` query param | mitigate | `Math.max(0, Number(...))` clamps negative offsets. Limit is fixed at 12 server-side — client cannot request unlimited rows. |
| T-05-05 | Information Disclosure | `getAuctionFeed` | accept | Returns only `status="active"` auctions. Business location (lat/lng) is not returned to client — only computed `distanceMiles` value. |
</threat_model>
